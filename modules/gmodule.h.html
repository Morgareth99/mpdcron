 28  * MPDCRON_EVENT_PLAYER
 29  * MPDCRON_EVENT_MIXER

 35 #include <glib.h>
 36 #include <mpd/client.h>
 37
 38 enum mpdcron_init_retval {
 39         MPDCRON_INIT_RETVAL_SUCCESS = 0,
 40         MPDCRON_INIT_RETVAL_FAILURE,
 41 };
 42
 43 enum mpdcron_run_retval {
 44         MPDCRON_RUN_RETVAL_SUCCESS = 0, /** Success **/
 45         MPDCRON_RUN_RETVAL_RECONNECT, /** Schedule a reconnection to mpd server **/
 46         MPDCRON_RUN_RETVAL_RECONNECT_NOW, /** Schedule a reconnection to mpd server immediately. **/
 47         MPDCRON_RUN_RETVAL_UNLOAD, /** Unload the module **/
 48 };
 49
 50 #ifndef MPDCRON_INTERNAL
 51 /**
 52  * Initialize function. This function is called when the module is loaded.
 53  *
 54  * @param nodaemon Non-zero if --no-daemon option is passed to mpdcron.
 55  * @param fd This file descriptor points to mpdcron's configuration file.
 56  *           Don't call g_key_file_free() on this!
 57  * @return @see enum mpdcron_init_retval
 58  */
 59 int mpdcron_init(int nodaemon, GKeyFile *fd);
 60
 61 /**
 62  * Close function. This function is called when the module is unloaded.
 63  */
 64 void mpdcron_close(void);
 65
 66 #if defined(MPDCRON_EVENT_DATABASE)
 67 /**
 68  * Run function for database event
 69  */
 70 int mpdcron_run(const struct mpd_connection *conn, const struct mpd_stats *stats);
 71 #elif defined(MPDCRON_EVENT_STORED_PLAYLIST)
 72 /**
 73  * Run function for stored playlist event
 74  */
 75 int mpdcron_run(const struct mpd_connection *conn);
 76 #elif defined(MPDCRON_EVENT_QUEUE)
 77 /**
 78  * Run function for queue event
 79  */
 80 int mpdcron_run(const struct mpd_connection *conn);
 81 #elif defined(MPDCRON_EVENT_PLAYER)
 82 /**
 83  * Run function for player event
 84  */
 85 int mpdcron_run(const struct mpd_connection *conn, const struct mpd_song *song, const struct mpd_status *status);
 86 #elif defined(MPDCRON_EVENT_MIXER)
 87 /**
 88  * Run function for mixer event
 89  */
 90 int mpdcron_run(const struct mpd_connection *conn, const struct mpd_status *status);
 91 #elif defined(MPDCRON_EVENT_OUTPUT)
 92 /**
 93  * Run function for output event
 94  */
 95 int mpdcron_run(const struct mpd_connection *conn);
 96 #elif defined(MPDCRON_EVENT_OPTIONS)
 97 /**
 98  * Run function for options event
 99  */
100 int mpdcron_run(const struct mpd_connection *conn, const struct mpd_status *status);
101 #elif defined(MPDCRON_EVENT_UPDATE)
102 /**
103  * Run function for update event
104  */
105 int mpdcron_run(const struct mpd_connection *conn, const struct mpd_status *status);
106 #else
107 #error None of the required MPDCRON_EVENT_* constants were defined!
108 #endif
109
110 #endif /* !MPDCRON_INTERNAL */
111
112 #endif /* !MPDCRON_GUARD_MODULE_H */
